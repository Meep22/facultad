PIO EQU 30h

ORG 1000h
SECRET db 00110011b
FIN db "GANASTE"

ORG 3000h
poll: IN AL, PIO
CMP AL, SECRET
JZ ganaste
XOR AL, SECRET
NOT AL
OUT PIO + 1, AL
JMP poll

ganaste: OUT PIO + 1, AL
MOV BX, offset FIN
MOV AL, 7
INT 7
RET

ORG 2000h
MOV AL, 0
OUT PIO + 3, AL
MOV AL, 0FFh
OUT PIO + 2, AL
CALL poll
INT 0
end
