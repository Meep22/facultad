PIO EQU 30h

ORG 1000h
MSG db "ACTIVADO"

ORG 3000h
poll: MOV AH, 0
entrada: IN AL, PIO
CMP AL, 10000001b
JNZ poll
CMP AH, 0
JZ mensaje
JMP entrada

mensaje: MOV AH, 1
MOV BX, offset MSG
MOV AL, 8
INT 7
JMP entrada

ORG 2000h
MOV AL, 0
OUT PIO + 3, AL
MOV AL, 0FFh
OUT PIO + 2, AL
CALL poll
INT 0
end
