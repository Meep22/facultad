EOI EQU 20h
IMR EQU 21h
INT0 EQU 24h
N_F10 EQU 10

ORG 40
IP_F10 dw RUT_F10

ORG 1000h
MSG db "Vamos las interrupciones!"
FIN_STR db ?
CANT db 5

ORG 3000h
RUT_F10: PUSH AX
PUSH BX
MOV BX, offset MSG
MOV AL, offset FIN_STR - offset MSG
INT 7
MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
DEC CANT
JNZ FIN_INT
CLI
FIN_INT: IRET

ORG 2000h
CLI
MOV AL, 0FEh
OUT IMR, AL
MOV AL, N_F10
OUT INT0, AL
STI
loop: CMP CANT, 0
JZ FIN_PROGRAMA
JMP loop
FIN_PROGRAMA: INT 0
end
