CONT EQU 10h
COMP EQU 11h

EOI EQU 20h
IMR EQU 21h
INT1 EQU 25h
N_TIMER EQU 10

ORG 40
IP_TIMER dw RUT_TIMER

ORG 1000h
MSG db "Vamos las interrupciones!"
FIN_STR db ?
FIN_PROGRAMA db 0
CANT db 5

ORG 3000h
RUT_TIMER: PUSH AX
PUSH BX
MOV BX, offset MSG
MOV AL, offset FIN_STR - offset MSG
INT 7
MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
MOV FIN_PROGRAMA, 1
IRET

init: CLI
MOV AL, N_TIMER
OUT INT1, AL
MOV AL, 10
OUT COMP, AL
MOV AL, 0
OUT CONT, AL
MOV AL, 11111101b
OUT IMR, AL
STI
ret

ORG 2000h
CALL init
loop: CMP FIN_PROGRAMA, 1
JNZ loop
INT 0
end
