CONT EQU 10h
COMP EQU 11h

N_TIMER EQU 10
EOI EQU 20h
IMR EQU 21h
INT1 EQU 25h

ORG 40
dw RUT_TIMER

ORG 1000h
ESPERA db 1
STR db "Vamos las interrupciones!"
FIN_STR db ?
FIN_PROGRAMA db 0

ORG 3000h
RUT_TIMER: PUSH AX
PUSH BX
MOV AL, offset FIN_STR - offset STR
MOV BX, offset STR
INT 7
MOV AL, EOI
OUT EOI, AL
INC ESPERA
JZ FIN
MOV AL, 0
OUT CONT, AL
MOV AL, ESPERA
OUT COMP, AL
POP BX
POP AX
IRET
FIN: MOV FIN_PROGRAMA, 1
POP BX
POP AX
IRET

init: CLI
MOV AL, N_TIMER
OUT INT1, AL
MOV AL, ESPERA
OUT COMP, AL
MOV AL, 11111101b
OUT IMR, AL
MOV AL, 0
OUT CONT, AL
STI
RET

ORG 2000h
CALL init
loop: JMP loop
INT 0
end
