CONT EQU 10h
COMP EQU 11h

EOI EQU 20h
IMR EQU 21h
INT0 EQU 24h
INT1 EQU 25h
N_TIMER EQU 10
N_F10 EQU 11

ORG 40
dw RUT_TIMER
ORG 44
dw RUT_F10

ORG 1000h
VALOR db ?," "
FIN_PROGRAMA db 0


ORG 3000h
RUT_TIMER: PUSH AX
PUSH BX
MOV BX, offset VALOR
MOV AL, 2
INT 7
DEC VALOR
CMP VALOR, 2Fh
JZ FIN
MOV AL, 0
OUT CONT, AL
MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
IRET
FIN: MOV FIN_PROGRAMA, 1
MOV AL, 0FFh
OUT IMR, AL
POP BX
POP AX
IRET


RUT_F10: CLI
PUSH AX
MOV AL, 11111101b
OUT IMR, AL
MOV AL, 1
OUT COMP, AL
MOV AL, 0
OUT CONT, AL
MOV AL, EOI
OUT EOI, AL
POP AX
STI
IRET

init: MOV AL, 11111110b
OUT IMR, AL
MOV AL, N_F10
OUT INT0, AL
MOV AL, N_TIMER
OUT INT1, AL
RET

ORG 2000h
MOV BX, offset VALOR
read_number: INT 6
cmp byte ptr [bx], '1'
js read_number
cmp byte ptr [bx], '9'+1
jns read_number
CALL init
loop: CMP FIN_PROGRAMA, 1
JNZ loop
INT 0
end
