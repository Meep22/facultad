CONT EQU 10h
COMP EQU 11h

EOI EQU 20h
IMR EQU 21h
INT0 EQU 24h
INT1 EQU 25h

N_F10 EQU 10
N_TIMER EQU 11

ORG 40
dw RUT_F10

ORG 44
dw RUT_TIMER

ORG 1000h
VALOR db "9 "
PAUSA db 0

ORG 3000h
RUT_F10: PUSH AX
CMP VALOR, '0'
JS reset
CMP PAUSA, 1
JZ reanudar
MOV AL, 11111110b
OUT IMR, AL
MOV PAUSA, 1
JMP fin_f10

reset: MOV VALOR, '9'
reanudar: MOV PAUSA, 0
MOV AL, 11111100b
OUT IMR, AL
MOV AL, 0
OUT CONT, AL

fin_f10: MOV AL, 0
OUT CONT, AL
MOV AL, EOI
OUT EOI, AL
POP AX
IRET

RUT_TIMER: PUSH AX
PUSH BX
MOV AL, 2
MOV BX, offset VALOR
INT 7
DEC VALOR
CMP VALOR, '0'
JS pausar
MOV AL, 0
OUT CONT, AL
fin_timer: MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
IRET

pausar: MOV PAUSA, 1
MOV AL, 11111110b
OUT IMR, AL
JMP fin_timer

init: PUSH AX
CLI
MOV AL, N_F10
OUT INT0, AL
MOV AL, N_TIMER
OUT INT1, AL
MOV AL, 11111100b
OUT IMR, AL
MOV AL, 1
OUT COMP, AL
MOV AL, 0
OUT CONT, AL
STI
POP AX
RET

ORG 2000h
CALL init
loop: JMP loop
end
