HAND EQU 40h

EOI EQU 20h
IMR EQU 21h
INT2 EQU 26h
N_HAND EQU 10

ORG 40
dw RUT_HAND

ORG 1000h
STR db "UNIVERSIDAD NACIONAL DE LA PLATA"
END_STR db ?
i db 0

ORG 3000h
RUT_HAND: PUSH AX
PUSH BX
MOV BX, offset STR
MOV AL, [BX]
ADD AL, i
OUT HAND, AL
IN AL, HAND + 1
OR AL, 00000010b
OUT HAND + 1, AL
MOV AL, EOI
OUT EOI, AL
INC BX
CMP BX, offset END_STR
JZ fin_hand
POP BX
POP AX
IRET

fin_hand: MOV AL, 0FFh
OUT IMR, AL
POP BX
POP AX
IRET

init: PUSH AX
MOV AL, 10000000b
OUT HAND + 1, AL
MOV AL, N_HAND
OUT INT2, AL
MOV AL, 11111011b
OUT IMR, AL
POP AX
RET

ORG 2000h
CALL init
loop: JMP loop
end
