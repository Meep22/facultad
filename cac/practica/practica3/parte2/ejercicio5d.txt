HAND EQU 40h

EOI EQU 20h
IMR EQU 21h
INT0 EQU 24h
INT2 EQU 26h
N_HAND EQU 10
N_F10 EQU 11

ORG 40
IP_HAND dw RUT_HAND

ORG 44
IP_F10 dw RUT_F10

ORG 1000h
STR db "          "
i dw 0

ORG 3000h
RUT_HAND: PUSH AX
PUSH BX
MOV BX, offset STR
ADD BX, i
MOV AL, [BX]
OUT HAND, AL
INC i
CMP i, 10
JZ fin_hand
MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
IRET

fin_hand: MOV AL, 11111111b
OUT IMR, AL
MOV AL, 0
OUT HAND + 1, AL
MOV AL, EOI
OUT EOI, AL
POP BX
POP AX
IRET

RUT_F10: PUSH AX
MOV AL, 11111111b
OUT IMR, AL
MOV AL, 0
OUT HAND + 1, AL
MOV AL, EOI
OUT EOI, AL
POP AX
IRET

init: PUSH AX
CLI
MOV AL, N_HAND
OUT INT2, AL
MOV AL, 10000000b
OUT HAND + 1, AL
MOV AL, N_F10
OUT INT0, AL
MOV AL, 11111010b
OUT IMR, AL
STI
POP AX
RET

ORG 2000h
MOV AH, 10
MOV BX, offset STR
leo: INT 6
INC BX
DEC AH
JNZ leo
CALL init
loop: JMP loop
end
